47 void Cjt_clusters::recalcular_distancia_wpgma(const pair<string,string>& min_dist, map<string,double>& dist_nou_cluster){
48  string id_nou_cluster = min_dist.first + min_dist.second;
49  for(map<string,dist_cluster>::iterator it = distancies.begin(); it != distancies.end(); ++it){
50  double dist;
51  string id_actual = it->first;
52  if(id_actual != min_dist.first and id_actual != min_dist.second){
53  //si l'id de l'espècie actual és més petit q l'id del cluster nou més
54  //petit vol dir que conté les distàncies amb els clusters que formen el nou cluster
55  if(id_actual < min_dist.first){
56  dist = (it->second.find(min_dist.first)->second+it->second.find(min_dist.second)->second)/2;
57  it->second.erase(min_dist.first);
58  it->second.erase(min_dist.second);
59  it->second.insert(make_pair(id_nou_cluster,dist));
60  }
61  //si l'id actual és més petit que l'id més gran del nou cluster vol dir que
62  //conté les seves distàncies
63  else if(id_actual < min_dist.second){ //
64  dist = it->second.find(min_dist.second)->second;
65  it->second.erase(min_dist.second);
66  auto aux = dist_nou_cluster.find(id_actual);
67  dist += aux->second;
68  dist /= 2;
69  aux->second = dist;
70  }
71  }
72  else if(id_actual == min_dist.first){
73  for(auto it1 = it->second.begin(); it1 != it->second.end(); ++it1){
74  if(it1->first != min_dist.second){
75  dist_nou_cluster.insert(make_pair(it1->first,it1->second));
76  }
77  }
78  }
79  else if(id_actual == min_dist.second){
80  for(auto it1 = it->second.begin(); it1 != it->second.end(); ++it1){
81  auto aux = dist_nou_cluster.find(it1->first);
82  dist = (aux->second + it1->second)/2;
83  aux->second = dist;
84  }
85  }
86  }
